function getAdd(){
	let args = []
	let inner_add = (...args)=>{
		return args.reduce((a,b)=>a+b)
	} 
	return function(...newArgs){
		if(Array.from(newArgs).length > 0){
			args.push(...newArgs)
		}
		else{
			inner_add(...args)
		}
	}
}
